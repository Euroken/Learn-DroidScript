Create Progress

app.LoadPlugin( "AlertPlus" );

function OnStart()
{
  lay = app.CreateLayout( "Linear", "VCenter,FillXY" );

  ap = app.CreateAlertPlus();

  ps = ap.CreateProgress( "Please Wait", "Hello", ap.THEME_LIGHT );
  ps.SetCancelable( false );
  ps.Show();

  for( var i=0; i<=100; i++ )
  {
  ps.UpdateProgress( i );
  if( i == 100 ) ps.Dismiss();
  }

  app.AddLayout( lay );
}
