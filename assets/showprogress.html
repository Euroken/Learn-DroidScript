app.LoadPlugin( "AlertPlus" );

function OnStart()
{
  lay = app.CreateLayout( "Linear", "VCenter,FillXY" );

  ap = app.CreateAlertPlus();

  prg = ap.ShowProgress( "Please Wait 3 seconds", "MyTitle", false, ap.THEME_LIGHT );
  setTimeout( function(){ ap.HideProgress(prg); app.ShowPopup( "End" ); }, 3000 );

  app.AddLayout( lay );
}
