Documentation for sjgTools

In order to use sjgTools, you must first load the plugin at the top of your script using the LoadPlugin method like this:

 app.LoadPlugin( "sjgTools" );

Then you can create an instance of the plugin object when you need it like this:

 plg = app.CreateObject( "sjgTools" );

Available functions

plg.createTempIcon( overwrite )

overwrite should be true or false

if overwrite is true, this function will replace an existing icon.

  •  Example - Create temp icon


function OnStart()
{
  lay = app.CreateLayout( "Linear", "VCenter,FillXY" );

  btn = app.CreateButton( "Button" );
  lay.AddChild( btn );

  app.AddLayout( lay );

  app.LoadPlugin( "sjgTools" );
  var plg = app.CreateObject( "sjgTools" );
  plg.createTempIcon( false );

}

plg.createMenu( list )

list is a comma separated list of values

please note your app must have function OnMenu() for this function to work

  •  Example - Create menu list


app.LoadPlugin("sjgTools");
var mnu;
function OnStart()
{
  lay = app.CreateLayout( "Linear", "VCenter,FillXY" );

  btn = app.CreateButton( "Button" );
  lay.AddChild( btn );

  app.AddLayout( lay );

  var plg = app.CreateObject( "sjgTools" );
  mnu=plg.createMenu( "Exit,Open" );
  mnu.Show();
}

//Handle list item selection.
function OnMenu( item )
{ 
        var temp;
    //Hide the dialog window.
    mnu.Hide();
    if(item==null)
    {
        mnu.Show();
        return;
    }
    switch(item) 
    { 
      case "Exit":
        app.Exit();
        break;
      case "Open":
        alert("Open");
        break;
default:
        app.Alert(item+" is not a valid option");
    } 
}
